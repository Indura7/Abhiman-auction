function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.openById(data.sheetId); var sh = ss.getSheetByName('Items') || ss.insertSheet('Items'); if (data.action === 'init' || data.action === 'write') { sh.clearContents(); sh.appendRow(['ID','Name','Brand','Notes','Cost','SoldPrice']); if (data.items) data.items.forEach(function(it){ sh.appendRow([it.id,it.name,it.brand,it.notes,it.cost,it.soldPrice===null?'':it.soldPrice]); }); } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var ss = SpreadsheetApp.openById(e.parameter.sheetId); var sh = ss.getSheetByName('Items'); if (!sh) return ContentService.createTextOutput(JSON.stringify({items:[]})).setMimeType(ContentService.MimeType.JSON); var rows = sh.getDataRange().getValues(); var items = []; for (var i=1;i<rows.length;i++){ items.push({id:rows[i][0],name:rows[i][1],brand:rows[i][2],notes:rows[i][3],cost:rows[i][4],soldPrice:rows[i][5]===''?null:rows[i][5]}); } return ContentService.createTextOutput(JSON.stringify({items:items})).setMimeType(ContentService.MimeType.JSON); }
